function isEqualStack(stack1, stack2)
{
    const temp = [];
    for (const element of stack1)
    {
        temp.push(element);
    }
    for (const element of stack2)
    {
        const i = temp.indexOf(element);
        if (i === -1) return false;
        temp.splice(i, 1);
    }
    return temp.length <= 0;
